local script = G2L["8e"];
	local Players = game:GetService("Players")
	local UserInputService = game:GetService("UserInputService")
	local player = Players.LocalPlayer
	
	local playerGui = player:WaitForChild("PlayerGui")
	local StandProudGUI = playerGui:WaitForChild("StandProudGUI")

	local Pages = MainFrame:WaitForChild("Pages")
	local MultiBarragePage = Pages:WaitForChild("MultiBarragePage")
	
	local OnKeyInput = MultiBarragePage:WaitForChild("OnKeyInputLabel")
	local OffKeyInput = MultiBarragePage:WaitForChild("OffKeyInputLabel")
	local StartButton = MultiBarragePage:WaitForChild("StartBarrageScriptButton")
	local StopButton = MultiBarragePage:WaitForChild("StopBarrageScriptButton")
	
	local AbleToBarrage = false
	local currentOnKey = nil
	local currentOffKey = nil
	
	local function textToKeyCode(text)
		if not text or text == "" or text == "Key here" then
			return nil
		end
		
		text = string.upper(string.gsub(text, "%s", ""))
		
		local success, keyCode = pcall(function()
			return Enum.KeyCode[text]
		end)
		
		if success and keyCode then
			return keyCode
		end
		
		return nil
	end
	
	local function updateKeyBindings()
		local onKeyText = OnKeyInput.Text
		local offKeyText = OffKeyInput.Text
		
		currentOnKey = textToKeyCode(onKeyText)
		currentOffKey = textToKeyCode(offKeyText)
		
	end
	
	StartButton.MouseButton1Click:Connect(function()
		AbleToBarrage = true
	end)
	
	StopButton.MouseButton1Click:Connect(function()
		AbleToBarrage = false
	end)
	
	local function multiBarrage()
		local BarrageArgsMulti = {
			"Barrage",
			game:GetService("Players").LocalPlayer.Character,
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"):WaitForChild("AnimationController"),
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"),
			true
		}
	
		local argsBlockBarrage1 = {
			"Block",
			game:GetService("Players").LocalPlayer.Character,
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"):WaitForChild("AnimationController"),
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"),
			true
		}
		game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Action"):FireServer(unpack(argsBlockBarrage1))
		task.wait(0.1)
		local argsBlockBarrage2 = {
			"Block",
			game:GetService("Players").LocalPlayer.Character,
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"):WaitForChild("AnimationController"),
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"),
			false
		}
		game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Action"):FireServer(unpack(argsBlockBarrage2))
	
		for countBarrage = 0,100,1 do
			game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Action"):FireServer(unpack(BarrageArgsMulti))
		end
	end
	
	local function StopmultiBarrage()
		local StopMultiBarrageArgs = {
			"Barrage",
			game:GetService("Players").LocalPlayer.Character,
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"):WaitForChild("AnimationController"),
			game:GetService("Players").LocalPlayer.Character:WaitForChild("Stand"),
			false
		}
		for countBarrage = 0,111,1 do
			game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Action"):FireServer(unpack(StopMultiBarrageArgs))
		end
	end
	
	local function onInputBegan(input, gameProcessed)
		if gameProcessed then return end
		if input.UserInputType ~= Enum.UserInputType.Keyboard then return end
		
		if not AbleToBarrage then
			return
		end
		
		if input.KeyCode == currentOnKey then
			multiBarrage()
			
		elseif input.KeyCode == currentOffKey then
			StopmultiBarrage()
		end
	end
	
	UserInputService.InputBegan:Connect(onInputBegan)
	
	OnKeyInput.Changed:Connect(updateKeyBindings)
	OffKeyInput.Changed:Connect(updateKeyBindings)
	
	updateKeyBindings()
	
	
end;
